\chapter{Derivaatta}

\laatikko[Derivointisäännöt]{
	\begin{description}
		\item[Summan derivaatta] $(f+g)'(x) = (f'+g')(x)$
		\item[Tulon derivaatta] $(fg)'(x) = (f'g+fg')(x)$
		\item[Osamäärän derivaatta] $(\frac{f}{g})'(x) = (\frac{f'g-fg'}{g^2})(x)$
	\end{description}
}

\laatikko[Potenssifunktion derivaatta]{
	$\diff{x^n} = nx^{n-1}$
}
